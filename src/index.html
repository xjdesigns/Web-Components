Icon Buttons


  <icon-button icon="check" size="sm" material text="icon button small"></icon-button>
  <icon-button icon="accessible" material></icon-button>

  // [text] attribute = text for the button
  // [icon] attribute = icon for the button, accepts a google ligature
  // [size] attribute = sm or lg, normal is the default
  // [material] attribute = add for material effect on hover
